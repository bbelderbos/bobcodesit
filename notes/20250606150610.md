# Effortless Grouping with defaultdict + Geo Mapping üåç

Want to group countries by continent using their 2-letter codes?

‚úÖ Use pycountry-convert for mapping.
‚úÖ Use defaultdict(list) to skip key checks.
‚úÖ pprint or pp() for a clean output.

```
# /// script
# dependencies = [
#   "pycountry",
#   "pycountry-convert",
# ]
# ///
import collections as co
from pprint import pp
import pycountry_convert as pc


countries = {
    "AU",  # Australia
    "BR",  # Brazil
    "CA",  # Canada
    "CN",  # China
    "DE",  # Germany
    "FR",  # France
    "GB",  # United Kingdom
    "IN",  # India
    "JP",  # Japan
    "KR",  # South Korea
    "MX",  # Mexico
    "NG",  # Nigeria
    "RU",  # Russia
    "US",  # United States
    "ZA",  # South Africa
}

results = co.defaultdict(list)
for country in countries:
    continent = pc.country_alpha2_to_continent_code(country)

    results[continent].append(country)

pp(results)

"""
Output:
defaultdict(<class 'list'>,
            {'AF': ['NG', 'ZA'],
             'AS': ['IN', 'CN', 'JP', 'KR'],
             'NA': ['MX', 'CA', 'US'],
             'EU': ['RU', 'DE', 'FR', 'GB'],
             'SA': ['BR'],
             'OC': ['AU']})
"""
```
