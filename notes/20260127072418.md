# Create GitHub issues faster from the CLI

Combine `gh`, `fzf`, and your `$EDITOR` to create issues without leaving the terminal:

```bash
ghissue() {
  local user tmp

  user=$(gh api repos/:owner/:repo/assignees --jq '.[].login') || return 1
  user=$(echo "$user" | fzf --prompt="Assign to: ")
  [[ -z "$user" ]] && return 1

  tmp=$(mktemp)
  trap 'rm -f "$tmp"' EXIT

  cat <<EOF > "$tmp"
@$user

Describe the issue here...
EOF

  ${EDITOR:-vim} "$tmp"
  gh issue create --assignee "$user" --body-file "$tmp"
}
```
