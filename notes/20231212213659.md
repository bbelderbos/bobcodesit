# Beautiful is better than ugly.

```
# Ugly -> less Pythonic, unnecessarily verbose
data = {'a': 1, 'b': 2, 'c': 3}
keys = data.keys()
values = []
for key in keys:
    values.append(data[key])
total = 0
for value in values:
    total += value
print(total)

# Beautiful -> Pythonic, concise and clear
data = {'a': 1, 'b': 2, 'c': 3}
total = sum(data.values())
print(total)
```
